MySQL/SQLite database access

To access to a MySQL/SQLite database it is necessary to add in the project (.pro)

   QT += sql
   QTPLUGIN += qsqlite

then (for example in automation.cpp) something like (error check missing):

  #include <QtSql>
  QSqlDatabase db;

In setup()

   db = QSqlDatabase::addDatabase("QMYSQL", "DATABASE_ID");  // or "QSQLITE"
   db.setHostName("DATABASE_ADDRESS"); 
   db.setDatabaseName("DATABASE_NAME"); 
   db.setUserName("DATABASE_USERNAME"); 
   db.setPassword("DATABASE_PASSWORD");
   db.open();
}

In loop()

{
   QString testo = "SELECT * FROM TABELLA_X" ;
   QSqlQuery query(testo, db); 
   testo += "\n";
    if (!query.exec()) { 
       testo += QString("Open error [%1]").arg(db.lastError().text());
    } else { 
       for (int n = 0; n < query.record().count(); n++)
       { 
           testo += query.record().fieldName(n) + " - "; 
       } 
       testo += "\n";
	
       while(query.next()) { 
               for (int n = 0; n < query.record().count(); n++) 
               { 
                   testo += query.value(n).toString() + " | ";
	
               } 
               testo += "\n";
       } 
   }
   query.finish(); 
   // in testo there is the result
}

Moreover, it is necessary to add on TPAC a folder “sqldrivers”  containing symbolic links:

	mkdir sqldrivers
	cd sqldrivers
	ln –s /usr/lib/libsqlite* ./
